{{- $img := .page.Resources.GetMatch .src -}} {{- if not $img }} {{- warnf "No image resource found at %s%s" .page.RelPermalink .src -}} {{- end -}} {{- $densities := .densities | default (slice 1 2) -}} {{- $src := "" -}} {{- $srcset := slice -}} {{- range $index, $density := $densities -}} {{- $width := mul $.width $density -}} {{- if gt $width $img.Width -}} {{- warnf "Source image %s is not wide enough to resize to %dpx (%dx density)" $img.RelPermalink $width $density -}} {{- end -}} {{- $resized := $img.Resize (printf "%dx" $width) -}} {{- $srcset = $srcset | append (printf "%s %dx" $resized.RelPermalink $density) -}} {{- if eq $density 1 -}} {{- $src = $resized -}} {{- end -}} {{- end -}} {{- $srcset = delimit $srcset ", " -}}
